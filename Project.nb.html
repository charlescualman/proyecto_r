# Funcion que evalua impacto - presupuesto - productividad
clasificar_proyecto <- function(impacto, presupuesto, productividad) {
  
  # Regla 1: Alto impacto y alta productividad
  if (impacto == "Alto" && productividad >= 1.5) {
    return("Proyecto estratégico")
    
  # Regla 2: Alto presupuesto y baja productividad
  } else if (presupuesto > 1000000 && productividad < 0.75) {
    return("Riesgo de ineficiencia")
    
  # Regla 3: Productividad media (entre 0.75 y 1.5)
  } else if (productividad >= 0.75 && productividad < 1.5) {
    return("Proyecto estándar")
    
  # Regla 4: Resto
  } else {
    return("Proyecto emergente")
  }
}

# Se aplica mapply dentro de mutate
proyectoInovacion <- proyectoInovacion %>%

    )
  )

#Función 1: Calcular índice de eficiencia del proyecto

# función que calcula la eficiencia para cada proyecto
fn_calculo_eficiencia <- function(pub, pat, pres) {
  return((pub + pat) / pres)
}

# Usamos mutate junto con lapply
# lapply devuelve una lista, por eso usamos unlist() para 
# que se convierta en una columna (vector) de nuestro dataframe.
proyectoInovacion <- proyectoInovacion %>%
  mutate(
    eficiencia_proyecto = unlist(lapply(1:n(), function(i) {
      fn_calculo_eficiencia(publicaciones[i], patentes[i], presupuesto_usd[i])
    }))
  )

# Recibe un dataframe y devuelve solo IDs y Categoría de los proyectos de Alto Impacto
identificar_proyectos_impacto <- function(df) {
  
  # ac solo los de alto impacto
  df_filtrado <- df %>%
    filter(impacto == "Alto") %>%
    #solo id_proyecto y la categoria del proyecto

# Recibe un dataframe y el nombre de la columna para analizar
generar_resumen_estadistico <- function(df, columna) {
  datos <- df[[columna]]
  
  resumen <- list(
    variable = columna,
    media = mean(datos),
    mediana = median(datos),
    desviacion_estandar = sd(datos),
    percentiles = quantile(datos, probs = c(0.25, 0.5, 0.75, 0.9))
  )
  return(resumen)
}
